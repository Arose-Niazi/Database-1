	empID INT UNSIGNED PRIMARY KEY,
	fName VARCHAR(20) NOT NULL,

	dateOfBirth DATE,
	position ENUM('Manager','Supervisor') DEFAULT('Manager'),

	email VARCHAR(128) UNIQUE,
